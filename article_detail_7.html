# java 문자열 치환 ```java //replace([기존문자],[바꿀문자]) String a ="문자를 어떻게 바꾼다고"; a = a.replace("어떻게", "이렇게"); System.out.println(a); // 출력값: 문자를 이렇게 바꾼다고 //replaceAll([정규식],[바꿀문자]) String a ="문자를 어떻게 바꾼다고"; a = a.replace("어떻게", "이렇게"); System.out.println(a); // 출력값: 문자를 이렇게 바꾼다고 //이렇게 보면 replace와 replaceAll이 똑같아 보입니다. // 하지만 둘은 차이점이 있습니다. //replace 사용 String a ="가.나.다.라.마.바.사"; a = a.replace(".", "/"); System.out.println(a); // replace는 인자값만을 변환해줍니다. 출력값 : 가/나/다/라/마/바/사 //replaceAll 사용 String a ="가.나.다.라.마.바.사"; a = a.replaceAll(".", "/"); System.out.println(a); // replaceAll은 정규식이기 때문에 .(마침표)가 모든 문자를 의미하기에 둘은 다릅니다. 출력값:///////////// // replaceFirst와 replace,replaceAll의 차이점 //replaceFirst 사용(첫번째 값만 바뀝니다. String a ="문자는 문자가 있어야지만 바뀌지"; a = a.replaceFirst("문자", "단어"); System.out.println(a); 출력값: 단어는 문자가 있어야지만 바뀌지 //replace replaceAll사용 String a ="문자는 문자가 있어야지만 바뀌지"; a = a.replace("문자", "단어"); System.out.println(a); 출력값: 단어는 단어가 있어야지만 바뀌지 ```
